DATATYPES PROGRAMS

1.Integer datatype:

i. program to add two numbers

#include<stdio.h>
int main(){
    int a=10,b=20,sum;
    sum=a+b;
    printf("Sum=%d",sum);
    return 0;
}

ii. program to find the product of two numbers

#include<stdio.h>
int main(){
    int a=5,b=4,product;
    product=a*b;
    printf("Product=%d",product);
    return 0;
}

iii. program to swap to numbers

#include<stdio.h>
int main(){
    int a=10,b=20,temp;
    temp=a;
    a=b;
    b=temp;
    printf("a=%d b=%d",a,b);
    return 0;
}


2.Float Datatype:

i. program to find the area of a circle

#include<stdio.h>
int main(){
    float r=5.5,area;
    area=3.14*r*r;
    printf("Area=%.2f",area);
    return 0;
}

ii. program to find the average of two numbers

#include<stdio.h>
int main(){
    float a=10.5,b=20.5,avg;
    avg=(a+b)/2;
    printf("Average=%.2f",avg);
    return 0;
}

iii. program to convert Celsius to fahrenheit

#include<stdio.h>
int main(){
    float c=37,f;
    f=(c*9/5)+32;
    printf("Fahrenheit=%.2f",f);
    return 0;
}

#include<stdio.h>
int main(){

}

ii. program to print the ascii value

#include<stdio.h>
int main(){

iii. program to check the vowel

#include<stdio.h>
int main(){
    char ch='a';
    if(ch=='a'||ch=='e'||ch=='i'||ch=='o'||ch=='u'){
       printf("Vowel");
      }
    else{
       printf("Not Vowel");
      }

}


4.Double Datatype:

i. program to find the simple interest

#include<stdio.h>
int main(){
    double p=10000,r=5,t=2,si;
    si=(p*r*t)/100;
    printf("Simple Interest=%.2lf",si);
    return 0;
}

ii. program to find the square root

#include<stdio.h>
#include<math.h>
int main(){
    double n=64,result;
    result=sqrt(n);
    printf("Square Root=%.2lf",result);
    return 0;
}

iii. program to display the power

#include<stdio.h>
#include<math.h>
int main(){
    double result;
    result=pow(2,3);
    printf("Power=%.2lf",result);
    return 0;
}
