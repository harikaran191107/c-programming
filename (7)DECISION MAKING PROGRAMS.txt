5. switch Statement:

i. Program 1

#include<stdio.h>
int main(){
    int day;
    printf("enter the day number:");
    scanf("%d",&day);
    switch(day){
    
     case 1:
     printf("Monday");
     break;

     case 2:
     printf("Tuesday");
     break;

     default:
     printf("Invalid");
    }
    return 0;
}


ii. Program 2

#include<stdio.h>
int main(){
    char op;
    printf("enter the operator:");
    scanf("%d",&op);
    switch(op){
     case '+':
     printf("Addition");
     break;

     case '-':
     printf("Subtraction");
     break;

     default:
     printf("Invalid operator");
    }
