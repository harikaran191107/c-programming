DECISION MAKING

i. simple if statement:

syntax:
if(condition){
   //statement;
  }

example:
#include <stdio.h>
int main(){
    int num;
    scanf("%d",&num);
    if (num > 0) {
        printf("The number is positive");
       }
    return 0;
   }

input:56
output:The number is positive


ii. else if statement:

syntax:
if(condition){
   //statement;
  }
else{
   //statement;
  }

example:
#include<stdio.h>
int main(){
    int num;
    scanf("%d",&num);
    if(num>0){
       printf("The number is positive");
      }
    else{
       printf("The number is negative");
      }
    return 0;
   }

input:-34
output:The number is negative


iii. else if ladder statement:

syntax:
if(condition){
   //statement;
  }
else if(condition){
   //statement;
  }
else if(condition){
   //statement;
  }
  . 
  .
  .
else{
   //statement;
  }

example:
#include<stdio.h>
int main(){
    int marks;
    scanf("%d", &marks);
    if(marks>=90){
        printf("Grade A");
    }
    else if(marks>=75){
        printf("Grade B");
    }
    else if(marks>=50){
        printf("Grade C");
    }
    else{
        printf("Fail");
    }
    return 0;
   }

input:97
output:Grade A


iv. nested if:

syntax:
if(condition1){
    if(condition2){
        //statement;
       }
   }

example:
#include<stdio.h>
int main(){
    int num;
    scanf("%d",&num);
    if(num>0){
       if(num%2==0) {
          printf("Positive even number");
         }
       else{
          printf("Positive odd number");
         }
    }
    else{
        printf("Number is zero or negative");
       }
    return 0;
   }

input:45
output:positive odd number


v. switch case:

syntax:
switch(varname){
    case value1:
        //statements
        break;

    case value2:
        //statements
        break;

    case value3:
        //statements
        break;

    default:
        //statements
   }

example:
#include<stdio.h>
int main(){
    int choice;
    scanf("%d",&choice);
    switch(choice){
        case 1:
            printf("You selected One");
            break;

        case 2:
            printf("You selected Two");
            break;

        case 3:
            printf("You selected Three");
            break;

        default:
            printf("Invalid choice");
       }
     return 0;
    }
 
input:3
output:You selected Three
