OPERATORS PROGRAMS

1.Arithmetic Operators

i. Program to Add Two Numbers

#include<stdio.h>
int main(){
    int a,b,sum;
    printf("enter the number 1:");
    scanf("%d",&a);
    printf("enter the number 2:");
    scanf("%d",&b);
    sum=a+b;
    printf("Sum=%d",sum);
    return 0;
}
